```Js
function solution(n, arr1, arr2) {
  // n인 정사각형
  // arr1, arr2 2진수로해서 벽은 # 아니면 ""
  let map1 = arr1.map((a) =>{
    if(a.toString(2).length !==5){
      return a.toString(2).padStart(5,"0")
    }else{
       return a.toString(2)
    }
  })
  let map2 = arr2.map((a)=>{
      if(a.toString(2).length !==5){
      return a.toString(2).padStart(5,"0")
    }else{
       return a.toString(2)
    }
  })
}
```
