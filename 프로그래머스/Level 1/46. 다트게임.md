  // s = 1제곱 d = 2제곱 t = 3제곱
  // * 은 2배 *2개면 4배 #은 - 너스 # 2개면 -2배

  // 길이는 6
  // 옵션이 1개 일 경우
  // 길이는 7
  // 옵션이 2개 일 경우
  // 길이는 8
  // 옵션이 3개 일 경우
  // 길이는 9
  let answer = 0;
  for (let i = 0; i < dartResult.length; i++) {
    if (isNaN(dartResult[i]) === false) {
      if (
        (dartResult[i] === "S") |
        (dartResult[i] === "D") |
        (dartResult[i] === "T")
      ) {
        answer += dartResult[i] ** dartResult[i + 1];
        if ((dartResult[i] === "*") | (dartResult[i] === "#")) {
        }
      }
    }
  }
  console.log(answer);
}
