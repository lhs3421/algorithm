function solution(dartResult) {
  let answer = 0;
  let result = [];
  for (let i = 0; i < dartResult.length; i++) {
    if (dartResult[i] >= 0 && dartResult[i] <= 9) {
      if (dartResult[i + 1] === "0") {
        answer += 10;
      } else {
        answer += Number(dartResult[i]);
      }
    } else if (dartResult[i] === "S") {
      if (dartResult[i + 1] === "*") {
        result.push(answer * 2);
        answer = 0;
      } else if (dartResult[i + 1] === "#") {
        result.push(answer * -1);
        answer = 0;
      } else {
        result.push(answer);
        answer = 0;
      }
    } else if (dartResult[i] === "D") {
      answer = answer ** 2;
      if (dartResult[i + 1] === "*") {
        result.push(answer * 2);
        answer = 0;
      } else if (dartResult[i + 1] === "#") {
        result.push(answer * -1);
        answer = 0;
      } else {
        result.push(answer);
        answer = 0;
      }
      answer = 0;
    } else if (dartResult[i] === "T") {
      answer = answer ** 3;
      if (dartResult[i + 1] === "*") {
        result.push(answer * 2);
        answer = 0;
      } else if (dartResult[i + 1] === "#") {
        result.push(answer * -1);
        answer = 0;
      } else {
        result.push(answer);
        answer = 0;
      }
    }
  }
  console.log(result);
  return result.reduce((a, b) => a + b, 0);
}
