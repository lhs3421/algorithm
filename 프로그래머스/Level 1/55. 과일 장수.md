# 과일 장수

## 나의 문제 풀이

```js
function solution(k, m, score) {
  const sort = score.sort((a, b) => b - a);
  let answer = 0;
  let box = [];
  const map = sort.map((a) => {
    if (box.length !== m) box.push(a);
    if (box.length === m) {
      const min = Math.min(...box);
      const sum = min * m;
      answer += sum;
      box = [];
    }
  });
  return answer;
}
```
