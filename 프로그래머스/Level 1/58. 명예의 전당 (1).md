## 나의 문제 풀이

```js
function solution(k, score) {
  let answer = [];
  let record = [];
  for (let i = 0; i < score.length; i++) {
    if (i < k) {
      record.push(score[i]);
      answer.push(Math.min(...record));
    }
    if (i >= k) {
      const min = Math.min(...record);
      if (min < score[i]) {
        const index = record.indexOf(min);
        record.splice(index, 1, score[i]);
        answer.push(Math.min(...record));
      } else {
        answer.push(Math.min(...record));
      }
    }
  }
  return answer;
}
```
