- 더 풀어봐야 함

```js
function solution(s) {
  let start = s;
  let answer = [0, 0];
  const binary = (string) => {
    const countDeleteZero = string
      .split("")
      .filter((a) => {
        if (a === "0") answer[1]++;
        return a;
      })
      .join("")
      .length.toString(2);
    answer[0]++;
    console.log(start);
    start = countDeleteZero;
  };
  while (start.includes("0")) {
    binary(start);
  }
  return answer;
}
```
